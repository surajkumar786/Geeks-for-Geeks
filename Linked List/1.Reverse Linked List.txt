class Solution 
{
  public:
    // Function to reverse a linked list.
    struct Node* reverseList(struct Node* head) 
    {
        Node* curr=head;
        Node* prev=NULL;
        
        while(curr)
        {
            Node* next=curr->next;
            curr->next=prev;
            prev=curr;
            curr=next;
        }
        
        return prev;
    }
};
