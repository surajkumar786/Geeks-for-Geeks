class Solution 
{
  public:
    // Function to remove duplicates from sorted linked list.
    Node* removeDuplicates(Node* head) 
    {

        if(head==NULL) return NULL;
        Node* temp=head;
        
        while(temp->next)
        {
            if(temp->data == temp->next->data)
            {
                temp->next=temp->next->next;
            }
            else
            {
                temp=temp->next;
            }
        }
        
        return head;
    }
};
